import express from 'express';
import cors from 'cors';
import dotenv from 'dotenv';
import Anthropic from '@anthropic-ai/sdk';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';

dotenv.config();

const __dirname = dirname(fileURLToPath(import.meta.url));
const app = express();
const PORT = process.env.PORT || 3001;
const MODEL = 'claude-sonnet-4-20250514';

if (!process.env.ANTHROPIC_API_KEY) {
  console.error('❌  ANTHROPIC_API_KEY missing — copy .env.example to .env and add your key.');
  process.exit(1);
}

const ai = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
app.use(cors({ origin: ['http://localhost:5173', 'http://localhost:4173'] }));
app.use(express.json());

// ─── Helpers ──────────────────────────────────────────────────────────────────

function parseJSON(text) {
  const cleaned = text.replace(/```json\s*/g, '').replace(/```\s*/g, '').trim();
  return JSON.parse(cleaned);
}

async function chat(system, userContent, maxTokens = 1024) {
  const params = {
    model: MODEL,
    max_tokens: maxTokens,
    messages: [{ role: 'user', content: userContent }],
  };
  if (system) params.system = system;
  const res = await ai.messages.create(params);
  return res.content[0].text;
}

/**
 * POST /api/ai/create-character
 * Generates backstory, traits, opening scene, starting items, and special ability.
 */
app.post('/api/ai/create-character', async (req, res) => {
  const { name, race, characterClass, background, stats } = req.body;
  try {
    const text = await chat(
      null,
      `You are a master fantasy RPG storyteller creating a vivid, unique character.
Character:
- Name: ${name}
- Race: ${race}
- Class: ${characterClass}
- Background: ${background}
- Stats: STR ${stats.strength} | INT ${stats.intelligence} | DEX ${stats.dexterity} | CHA ${stats.charisma} | WIS ${stats.wisdom}

Return ONLY valid JSON (no markdown, no preamble):
{
  "backstory": "3-4 sentences of compelling, specific backstory that references their race, class, and background",
  "personalityTraits": ["trait1", "trait2", "trait3"],
  "motivation": "one sentence — what drives this character more than anything",
  "flaw": "one specific character flaw",
  "startingLocation": "name of a unique starting town or area (be creative)",
  "openingNarrative": "3 sentences that drop the player into the opening scene — second person, present tense, vivid sensory detail",
  "startingItems": [
    { "name": "item1", "description": "short flavor text", "type": "weapon|armor|consumable|misc" },
    { "name": "item2", "description": "short flavor text", "type": "weapon|armor|consumable|misc" },
    { "name": "item3", "description": "short flavor text", "type": "weapon|armor|consumable|misc" }
  ],
  "specialAbility": { "name": "unique ability name", "description": "what it does in 1 sentence" }
}`,
      1200
    );
    res.json(parseJSON(text));
  } catch (err) {
    console.error('/create-character error:', err.message);
    res.status(500).json({ error: err.message });
  }
});

/**
 * POST /api/ai/npc-dialog
 * Full NPC conversation with persistent memory.
 */
app.post('/api/ai/npc-dialog', async (req, res) => {
  const { npc, character, playerMessage, conversationHistory } = req.body;
  try {
    const messages = [
      ...conversationHistory.map(h => ({ role: h.role, content: h.content })),
      { role: 'user', content: playerMessage }
    ];
    const params = {
      model: MODEL,
      max_tokens: 500,
      system: `You are ${npc.name}${npc.title ? ` "${npc.title}"` : ''} — ${npc.description}
Personality: ${npc.personality}
You are speaking to ${character.name}, a ${character.race} ${character.characterClass}.
Stay completely in character at all times. Keep responses to 2-3 sentences.
You may offer quests, sell items, share rumors, or react with hostility based on personality.
Never break the fourth wall. Never acknowledge you are an AI.`,
      messages
    };
    const aiRes = await ai.messages.create(params);
    res.json({
      response: aiRes.content[0].text,
      npcName: npc.name
    });
  } catch (err) {
    console.error('/npc-dialog error:', err.message);
    res.status(500).json({ error: err.message });
  }
});
